### 3.2 GLB/GLTF模型加载方案（Three.js）

```typescript
// 核心加载器配置
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js';
import { DRACOLoader } from 'three/addons/loaders/DRACOLoader.js';

// Draco压缩模型解码器（大幅减小模型体积）
const dracoLoader = new DRACOLoader();
dracoLoader.setDecoderPath('https://www.gstatic.com/draco/versioned/decoders/1.5.6/');

const gltfLoader = new GLTFLoader();
gltfLoader.setDRACOLoader(dracoLoader);

// 加载示例
async function loadModel(url: string, container: HTMLElement) {
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);
  const renderer = new THREE.WebGLRenderer({ antialias: true });
  
  renderer.setSize(container.clientWidth, container.clientHeight);
  container.appendChild(renderer.domElement);

  const gltf = await gltfLoader.loadAsync(url);
  scene.add(gltf.scene);
  
  // 自动适应视角
  const box = new THREE.Box3().setFromObject(gltf.scene);
  const center = box.getCenter(new THREE.Vector3());
  const size = box.getSize(new THREE.Vector3());
  camera.position.set(center.x + size.x, center.y + size.y, center.z + size.z * 2);
  camera.lookAt(center);
}
```

### 3.4 PubChem分子结构加载（3Dmol.js）

```typescript
// 3Dmol.js 分子展示
import $3Dmol from '3dmol';

async function showMolecule(container: HTMLElement, compoundName: string) {
  const viewer = $3Dmol.createViewer(container, {
    backgroundColor: 'white'
  });

  // 通过PubChem API获取3D结构
  const response = await fetch(
    `https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/name/${compoundName}/record/JSON?record_type=3d`
  );
  const data = await response.json();
  const conformer = data.PC_Compounds[0].atoms.conf[0].conformers[0];

  viewer.addModel(JSON.stringify(conformer), 'sdf');
  viewer.setStyle({}, { stick: { radius: 0.15 }, sphere: { scale: 0.3 } });
  viewer.zoomTo();
  viewer.render();
}

// 使用示例
showMolecule(document.getElementById('mol-viewer'), 'caffeine');
```

interface IframeEmbedConfig {
  phet: {

// PhET嵌入示例
function embedPhetSim(container: HTMLElement, simName: string, options?: {
  locale?: string;
  showControls?: boolean;
  capturePhysics?: boolean;
}) {
  const iframe = document.createElement('iframe');
  const lang = options?.locale || 'zh';
  iframe.src = `${config.phet.baseUrl}/${simName}/latest/${simName}_${lang}.html`;
  iframe.style.width = '100%';
  iframe.style.height = '600px';
  iframe.style.border = 'none';
  container.appendChild(iframe);
}

### 3.6 性能优化建议

| 优化项 | 方案 | 预期效果 |
|--------|------|----------|
| **3D模型压缩** | Draco算法压缩.glb | 体积减少70-90% |
| **纹理优化** | WebP格式纹理 / 2K上限 | 显存占用降低50% |
| **LOD层级** | Three.js LOD组件 | 远距离简化模型 |
| **异步加载** | GLTFLoader异步+进度条 | 用户体验提升 |
| **骨骼动画** | Drako解码器加速 | 加载速度提升3-5倍 |
| **帧率控制** | requestAnimationFrame + 自适应 | 电池寿命优化 |
| **预加载** | 模型预加载+缓存 | 切换无等待 |
| **内存管理** | 场景切换时dispose() | 防止内存泄漏 |

### 6.2 Nginx配置

```nginx
# nginx.conf
events {
    worker_connections 1024;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # Gzip压缩
    gzip on;
    gzip_types text/plain application/json application/javascript text/css application/vnd.ms-fontobject;
    gzip_min_length 1000;

    # 3D模型静态资源 - 长期缓存
    location ~* \.(glb|gltf)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # 图片资源 - 长期缓存
    location ~* \.(jpg|png|webp|svg)$ {
        expires 1y;
        add_header Cache-Control "public";
    }

    # 视频资源
    location ~* \.(mp4|webm)$ {
        expires 7d;
        add_header Cache-Control "public";
    }

    # API代理 - Node.js
    location /api/ {
        proxy_pass http://api:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
    }

    # Python FastAPI服务
    location /content/ {
        proxy_pass http://backend-fastapi:8000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
    }

    # 静态页面
    location / {
        root /usr/share/nginx/html;
        try_files $uri $uri/ /index.html;
    }
}
```
